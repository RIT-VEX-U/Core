244 template <typename Number> Number get_number() {
245 // ensures that the function is only used on numbers
246 static_assert(
247 std::is_floating_point<Number>::value || std::is_integral<Number>::value,
248 "This function should only be used on numbers"
249 );
250 // checks that the size of the number its trying to read combined with its location
251 // doesnt put it past the packet size
252 if (read_head + sizeof(Number) > pac.size()) {
253 printf(
254 "%s:%d: Reading a number[%d] at position %d would read past "
255 "buffer of "
256 "size %d\n",
257 __FILE__, __LINE__, sizeof(Number), read_head, pac.size()
258 );
259 return 0;
260 }
261 Number value = 0;
262 // copies the the number at the reader head to the Number's stored value and
263 // adds the size of the number to the read head so it moves on to the next set of bits
264 std::memcpy(&value, &pac[read_head], sizeof(Number));
265 read_head += sizeof(Number);
266 return value;
267 }
