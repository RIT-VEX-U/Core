228 void predict(const InputVector &u, double dt) {
229 // Our noise is continuous, so we need to discretize
230 const EMat<STATES, STATES> Q = sqrt_Q_ * sqrt(dt);
231
232 // Generate sigma points around the state mean
233 //
234 // equation (17)
235 EMat<STATES, NUM_SIGMAS> sigmas = pts_.square_root_sigma_points(xhat_, S_);
236
237 // Project each sigma point forward in time according to the
238 // dynamics f(x, u)
239 //
240 // sigmas = 𝒳ₖ₋₁
241 // sigmasF = 𝒳ₖ,ₖ₋₁ or just 𝒳 for readability
242 //
243 // equation (18)
244 for (int i = 0; i < NUM_SIGMAS; ++i) {
245 StateVector x = sigmas.template block<STATES, 1>(0, i);
246 sigmas_F_.template block<STATES, 1>(0, i) = integrator_(f_, x, u, dt);
247 }
248
249 // Pass the predicted sigmas (𝒳) through the Unscented Transform
250 // to compute the prior state mean and covariance
251 //
252 // equations (18) (19) and (20)
253 auto [xhat, S] = square_root_ut<STATES, STATES>(
254 sigmas_F_, pts_.Wm(), pts_.Wc(), mean_func_X_, residual_func_X_,
255 Q.template triangularView<Eigen::Lower>()
256 );
257
258 xhat_ = xhat;
259 S_ = S;

337 template <int ROWS>
338 void correct(
339 const InputVector &u, const EVec<ROWS> &y,
340 const std::function<EVec<ROWS>(const StateVector &, const InputVector &)> &h,
341 const EVec<ROWS> measurement_stddevs,
342 const std::function<EVec<ROWS>(const EMat<ROWS, NUM_SIGMAS> &, const EVec<NUM_SIGMAS> &)> &mean_func_Y,
343 const std::function<EVec<ROWS>(const EVec<ROWS> &, const EVec<ROWS> &)> &residual_func_Y,
344 const std::function<StateVector(const StateVector &, const StateVector &)> &residual_func_X,
345 const std::function<StateVector(const StateVector &, const StateVector &)> &add_func_X
346 ) {
347
348 EMat<ROWS, ROWS> sqrt_R = measurement_stddevs.asDiagonal();
349
350 // Generate new sigma points from the prior mean and covariance
351 // and transform them into measurement space using h(x, u)
352 //
353 // sigmas = 𝒳
354 // sigmasH = 𝒴
355 //
356 // This differs from equation (22) which uses
357 // the prior sigma points, regenerating them allows
358 // multiple measurement updates per time update
359 EMat<ROWS, NUM_SIGMAS> sigmas_H;
360 EMat<STATES, NUM_SIGMAS> sigmas = pts_.square_root_sigma_points(xhat_, S_);
361 for (int i = 0; i < NUM_SIGMAS; ++i) {
362 sigmas_H.template block<ROWS, 1>(0, i) = h(sigmas.template block<STATES, 1>(0, i), u);
363 }
364
365 // Pass the predicted measurement sigmas through the Unscented Transform
366 // to compute the mean predicted measurement and square-root innovation
367 // covariance.
368 //
369 // equations (23) (24) and (25)
370 auto [yhat, Sy] = square_root_ut<ROWS, STATES, NUM_SIGMAS>(
371 sigmas_H, pts_.Wm(), pts_.Wc(), mean_func_Y, residual_func_Y, sqrt_R.template triangularView<Eigen::Lower>()
372 );
373
374 // Compute cross covariance of the predicted state and measurement sigma
375 // points given as:
376 //
377 // L+1
378 // P_{xy} = Σ Wᵢ⁽ᶜ⁾[𝒳ᵢ - x̂][𝒴ᵢ - ŷ⁻]ᵀ
379 // i=0
380 //
381 // equation (26)
382 EMat<STATES, ROWS> Pxy;
383 Pxy.setZero();
384 for (int i = 0; i < NUM_SIGMAS; ++i) {
385 Pxy += pts_.Wc(i) * (residual_func_X(sigmas_F_.template block<STATES, 1>(0, i), xhat_)) *
386 (residual_func_Y(sigmas_H.template block<ROWS, 1>(0, i), yhat)).transpose();
387 }
388
389 // Compute the Kalman gain. We use Eigen's forward and backward substitution
390 // to solve. The equation in the paper uses MATLAB's / operator, but Eigen's
391 // solvers act like the \ operator, so we need to rearrange the equation to
392 // use those.
393 //
394 // K = (P_{xy} / S_{y}ᵀ) / S_{y}
395 // K = (S_{y} \ P_{xy})ᵀ / S_{y}
396 // K = (S_{y}ᵀ \ (S_{y} \ P_{xy}ᵀ))ᵀ
397 //
398 // equation (27)
399 EMat<STATES, ROWS> K = (Sy.transpose().template triangularView<Eigen::Upper>().solve(
400 Sy.template triangularView<Eigen::Lower>().solve(Pxy.transpose())
401 ))
402 .transpose();
403
404 // Compute the posterior state mean
405 //
406 // x̂ = x̂⁻ + K(y − ŷ⁻)
407 //
408 // second part of equation (27)
409 EVec<STATES> xhat_dot = K * residual_func_Y(y, yhat);
410 EVec<STATES> xhat = add_func_X(xhat_, xhat_dot);
411 EMat<STATES, STATES> S = S_;
412
413 // RECALIBRATE
414
415 // Add the change of xhat to each of the sigma points in 𝒳.
416 for (int i = 0; i < NUM_SIGMAS; i++) {
417 sigmas.template block<STATES, 1>(0, i) += (xhat_dot);
418 }
419
420 // Pass those sigma points through the measurement function to transform
421 // them into measurement space.
422 for (int i = 0; i < NUM_SIGMAS; ++i) {
423 sigmas_H.template block<ROWS, 1>(0, i) = h(sigmas.template block<STATES, 1>(0, i), u);
424 }
425
426 // Perform a second unscented transform, this time on the recalibrated
427 // measurement sigma points.
428 auto [yhat_k, Sy_k] = square_root_ut<ROWS, STATES, NUM_SIGMAS>(
429 sigmas_H, pts_.Wm(), pts_.Wc(), mean_func_Y, residual_func_Y, sqrt_R.template triangularView<Eigen::Lower>()
430 );
431
432 // Compute the cross covariance of the recalibrated sigma points.
433 Pxy.setZero();
434 for (int i = 0; i < NUM_SIGMAS; ++i) {
435 Pxy += pts_.Wc(i) * (residual_func_X(sigmas.template block<STATES, 1>(0, i), xhat)) *
436 (residual_func_Y(sigmas_H.template block<ROWS, 1>(0, i), yhat_k)).transpose();
437 }
438
439 // Compute the intermediate matrix U for downdating
440 // the square-root covariance
441 //
442 // equation (28)
443 const EMat<STATES, ROWS> U = K * Sy;
444
445 // Downdate the posterior square-root state covariance
446 //
447 // equation (29)
448 for (int i = 0; i < ROWS; i++) {
449 Eigen::internal::llt_inplace<double, Eigen::Lower>::rankUpdate(S, U.template block<STATES, 1>(0, i), -1);
450 }
451
452 // BACK OUT
453
454 // We only use the posterior state and covariance if it is more certain
455 // than the prior.
456 if ((S_ * S_.transpose()).trace() > (S * S.transpose()).trace()) {
457 xhat_ = xhat;
458 S_ = S;
459 }
460 }

500template <int COV_DIM, int STATES, int NUM_SIGMAS>
501std::tuple<EVec<COV_DIM>, EMat<COV_DIM, COV_DIM>> square_root_ut(
502 const EMat<COV_DIM, NUM_SIGMAS> &sigmas, const EVec<NUM_SIGMAS> &Wm, const EVec<NUM_SIGMAS> &Wc,
503 const std::function<EVec<COV_DIM>(const EMat<COV_DIM, NUM_SIGMAS> &, const EVec<NUM_SIGMAS> &)> &mean_func,
504 const std::function<EVec<COV_DIM>(const EVec<COV_DIM> &, const EVec<COV_DIM> &)> &residual_func,
505 const EMat<COV_DIM, COV_DIM> &sqrt_R
506) {
507 // New mean is usually just the sum of the sigmas * weights:
508 //
509 // L+1
510 // x̂ = Σ Wᵢ⁽ᵐ⁾𝒳ᵢ
511 // i=0
512 //
513 // equations (19) and (23) in the paper show this,
514 // but we allow a custom function, usually for angle wrapping
515 EVec<COV_DIM> x = mean_func(sigmas, Wm);
516
517 // Form an intermediate matrix S⁻ as:
518 //
519 // [√{W₁⁽ᶜ⁾}(𝒳_{1:L+1} - x̂) √{Rᵛ}]
520 //
521 // the part of equations (20) and (24) within the "qr{}"
522 EMat<COV_DIM, NUM_SIGMAS - 1 + COV_DIM> S_bar;
523 for (int i = 0; i < NUM_SIGMAS - 1; i++) {
524 S_bar.template block<COV_DIM, 1>(0, i) =
525 std::sqrt(Wc[1]) * residual_func(sigmas.template block<COV_DIM, 1>(0, i + 1), x);
526 }
527 S_bar.template block<COV_DIM, COV_DIM>(0, NUM_SIGMAS - 1) = sqrt_R;
528
529 // Compute the square-root covariance of the sigma points.
530 //
531 // We transpose S⁻ first because we formed it by horizontally
532 // concatenating each part; it should be vertical so we can take
533 // the QR decomposition as defined in the "QR Decomposition" passage
534 // of section 3. "EFFICIENT SQUARE-ROOT IMPLEMENTATION"
535 //
536 // The resulting matrix R is the square-root covariance S, but it
537 // is upper triangular, so we need to transpose it.
538 //
539 // equations (20) and (24)
540 EMat<COV_DIM, COV_DIM> S = S_bar.transpose()
541 .householderQr()
542 .matrixQR()
543 .template block<COV_DIM, COV_DIM>(0, 0)
544 .template triangularView<Eigen::Upper>()
545 .transpose();
546
547 // Update or downdate the square-root covariance with (𝒳₀-x̂)
548 // depending on whether its weight (W₀⁽ᶜ⁾) is positive or negative.
549 //
550 // equations (21) and (25)
551 Eigen::internal::llt_inplace<double, Eigen::Lower>::rankUpdate(
552 S, residual_func(sigmas.template block<COV_DIM, 1>(0, 0), x), Wc[0]
553 );
554
555 return std::make_tuple(x, S);
556}

651 void compute_weights(double alpha, double beta) {
652 double c = 1 / (alpha * alpha * (STATES + 1));
653 Wm_ = EVec<NUM_SIGMAS>::Constant(c);
654 Wc_ = EVec<NUM_SIGMAS>::Constant(c);
655
656 Wm_(0) = 1 - (1 / (alpha * alpha));
657 Wc_(0) = 1 - (1 / (alpha * alpha)) + (1 - alpha * alpha + beta);
658
659 EVec<STATES> t;
660 for (int i = 0; i < STATES; i++) {
661 t(i) = i + 1;
662 }
663
664 EVec<STATES> q = alpha * ((t * (STATES + 1)).cwiseQuotient(t + EVec<STATES>::Ones())).cwiseSqrt();
665
666 C_ = EMat<STATES, NUM_SIGMAS>::Zero();
667 for (int row = 0; row < STATES; row++) {
668 C_.row(row).segment(1, row + 1).setConstant(-q(row) / (row + 1));
669 }
670 C_.diagonal(2) = q;
671 }
