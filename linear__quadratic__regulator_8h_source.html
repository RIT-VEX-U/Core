94 ) {
95 auto [Ad, Bd] = discretize_AB(A, B, dt);
96
97 MatrixA S = DARE<STATES, INPUTS>(Ad, Bd, Q, R);
98
99 // (BᵀSD + R) \ (BᵀSA)
100 K_ = (Bd.transpose() * S * Bd + R).llt().solve(Bd.transpose() * S * Ad);
101 }
