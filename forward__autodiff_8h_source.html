225template <int X>
226EMat<X, X> hessian(const std::function<Dual(const EVecX<Dual, X>&)>& f, const EVec<X>& x) {
227 EMat<X, X> hess = EMat<X, X>::Zero();
228 const double h = 1e-6;
229
230 // For each component, compute the derivative of the gradient using finite differences of gradients computed with autodiff.
231 // This obviously isn't the most accurate, but it's easy and works better than doing it straight up numerically.
232 for (int i = 0; i < X; i++) {
233 EVec<X> x_plus = x;
234 EVec<X> x_minus = x;
235 x_plus(i) += h;
236 x_minus(i) -= h;
237
238 EVec<X> grad_plus = gradient<X>(f, x_plus);
239 EVec<X> grad_minus = gradient<X>(f, x_minus);
240
241 // The i-th column is the derivative of the gradient with respect to the i-th component.
242 hess.col(i) = (grad_plus - grad_minus) / (2 * h);
243 }
244
245 // Force symmetry.
246 hess = (hess + hess.transpose()) / 2.0;
247
248 return hess;
249}
