148template <int X>
149EMat<X, X> numerical_hessian(const std::function<double(const EVec<X> &x)> &f, const EVec<X> &x) {
150 EMat<X, X> hess = EMat<X, X>::Zero();
151 const double h = 1e-6;
152
153 // For each component, compute the derivative of the gradient.
154 for (int i = 0; i < X; i++) {
155 EVec<X> x_plus = x;
156 EVec<X> x_minus = x;
157 x_plus(i) += h;
158 x_minus(i) -= h;
159
160 EVec<X> grad_plus = numerical_gradient<X>(f, x_plus);
161 EVec<X> grad_minus = numerical_gradient<X>(f, x_minus);
162
163 // The i-th column is the derivative of the gradient with respect to the i-th component.
164 hess.col(i) = (grad_plus - grad_minus) / (2 * h);
165 }
166
167 // Force symmatry.
168 hess = (hess + hess.transpose()) / 2.0;
169
170 return hess;
