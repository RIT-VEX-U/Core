80
81 double x_factor = cos(theta_rad);
82 double y_factor = -sin(theta_rad);
83 double theta_factor = -(x * sin(theta_rad)) - (y * cos(theta_rad));
84
85 // prevent numerical error due to float precision
86 if (abs(y_factor) < 1e-9) {
87 y_factor = 0;
88 }
89 if (abs(x_factor) < 1e-9) {
90 x_factor = 0;
91 }
92
93 transfer_matrix.row(i) << x_factor, y_factor, theta_factor;

107 pose_t update() override {
108 Eigen::Vector<double, WHEELS> radian_deltas;
109
110 for (int i = 0; i < WHEELS; i++) {
111 double angle_rad = encoders[i].position(rev) * M_PI * 2;

117 pose_t updated_pos = calculate_new_pos(radian_deltas, current_pos);
118
119 // if we do not pass in an IMU we use the wheels for rotation
120 if (imu != nullptr) {
121 angle = 0;
122 // Translate "0 forward and clockwise positive" to "CCW positive and radians"
123 angle = -imu->rotation(vex::rotationUnits::rev) * 2 * M_PI;
124 // Offset the angle, if we've done a set_position
125 angle += angle_offset;
126 }

128 static pose_t last_pos = updated_pos;
129 static double last_speed = 0;
130 static double last_ang_speed = 0;
131 static timer tmr;
132
133 double speed_local = 0;
134 double accel_local = 0;
135 double ang_speed_local = 0;
136 double ang_accel_local = 0;
137 bool update_vel_accel = tmr.time(sec) > 0.1;
138
139 // This loop runs too fast. Only check at LEAST every 1/10th sec
140 if (update_vel_accel) {
141 // Calculate robot velocity
142 speed_local = pos_diff(updated_pos, last_pos) / tmr.time(sec);
143
144 // Calculate robot acceleration
145 accel_local = (speed_local - last_speed) / tmr.time(sec);
146
147 // Calculate robot angular velocity (deg/sec)
148 ang_speed_local = smallest_angle(updated_pos.rot, last_pos.rot) / tmr.time(sec);
149
150 // Calculate robot angular acceleration (deg/sec^2)
151 ang_accel_local = (ang_speed_local - last_ang_speed) / tmr.time(sec);
152
153 tmr.reset();
154 last_pos = updated_pos;
155 last_speed = speed_local;
156 last_ang_speed = ang_speed_local;
157 }
158
159 this->current_pos = updated_pos;
160 if (update_vel_accel) {
161 this->speed = speed_local;
162 this->accel = accel_local;
163 this->ang_speed_deg = ang_speed_local;
164 this->ang_accel_deg = ang_accel_local;
165 }

169 }

204 pose_t calculate_new_pos(Eigen::Vector<double, WHEELS> radian_deltas, pose_t old_pose) {
205 // Mr T = E -> Mr^{+} E = T
206 // We take the diagonal of radian_deltas to do a coefficient wise multiplication rather than a dot product
207 Eigen::Vector3d pose_delta = transfer_matrix_pseudoinverse * (radian_deltas.asDiagonal() * wheel_radii);
208 Eigen::Vector3d old_pose_vector{old_pose.x, old_pose.y, old_pose.rot};
209
210 // we achieve better performance by using the imu for rotation directly when possible
211 // If an imu is not passed in when constructing, simply use the wheels for rotation
212 if (imu != nullptr) {
213 pose_delta(2) = angle - old_angle;
214 }
215
216 pose_t new_pose = pose_exponential(old_pose_vector, pose_delta);
217
218 // simply replaces the calculated angle with the imu angle directly
219 if (imu != nullptr) {
220 new_pose.rot = angle;
221 }
222
223 return new_pose;
224 }
