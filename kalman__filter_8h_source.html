52 void predict(const InputVector &u) {
53 // Compute prior mean
54 // (Tracy 4)
55 m_xhat = m_F * m_xhat + m_B * u;
56
57 // Form temporary matrix to compute prior state covariance
58 EMat<STATES * 2, STATES> S_bar;
59 S_bar.template block<STATES, STATES>(0, 0) = (m_S * m_F.transpose());
60 S_bar.template block<STATES, STATES>(STATES, 0) = m_sqrt_Q;
61
62 // Compute prior state covariance
63 // (Tracy 5)
64 m_S = S_bar.householderQr()
65 .matrixQR()
66 .template block<STATES, STATES>(0, 0)

77 template <int ROWS>
78 void correct(const EVec<ROWS> &y,
79 const EMat<ROWS, STATES> &H,
80 const EMat<ROWS, ROWS> &R) {
81 // Compute measurement innovation
82 // (Tracy 8)
83 EVec<ROWS> z = y - (H * m_xhat);
84
85 // Form temporary matrix to compute innovation covariance
86 EMat<STATES + ROWS, ROWS> Sy_bar;
87 Sy_bar.template block<STATES, ROWS>(0, 0) = (m_S * H.transpose());
88 Sy_bar.template block<ROWS, ROWS>(STATES, 0) = R;
89
90 // Compute innovation covariance
91 // (Tracy 9)
92 EMat<ROWS, ROWS> Sy =
93 Sy_bar.householderQr()
94 .matrixQR()
95 .template block<ROWS, ROWS>(0, 0)
96 .template triangularView<Eigen::Upper>();
97
98 // Compute Kalman gain
99 // (Tracy 10)
100 EMat<STATES, ROWS> K =
101 (Sy.fullPivHouseholderQr().solve(
102 (Sy.transpose().fullPivHouseholderQr().solve(H)) *
103 (m_S.transpose() * m_S)))
104 .transpose();
105
106 // Compute posterior state
107 // (Tracy 13)
108 m_xhat = m_xhat + K * z;
109
110 // Form temporary matrix to compute posterior state covariance
111 EMat<STATES + ROWS, STATES> S_bar;
112 S_bar.template block<STATES, STATES>(0, 0) =
113 (m_S * (EMat<STATES, STATES>::Identity() - (K * H))
114 .transpose());
115 S_bar.template block<ROWS, STATES>(STATES, 0) = (R * K.transpose());
116
117 // Compute posterior state covariance
118 // (Tracy 14)
119 m_S = S_bar.householderQr()
120 .matrixQR()
121 .template block<STATES, STATES>(0, 0)
122 .template triangularView<Eigen::Upper>();
123 }
