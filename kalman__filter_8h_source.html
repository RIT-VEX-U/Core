143 void predict(const InputVector &u, const double &dt) {
144 // Q is discrete sqrt(process noise)
145 EMat<STATES, STATES> Q = m_sqrtQ * sqrt(dt);
146 auto [A, B] = discretize_AB(m_A, m_B);
147
148 // Compute prior mean
149 // equation (4)
150 m_xhat = A * m_xhat + B * u;
151
152 // Form temporary matrix to compute prior state covariance
153 EMat<STATES * 2, STATES> S_bar;
154 S_bar.template block<STATES, STATES>(0, 0) = (m_S * A.transpose());
155 S_bar.template block<STATES, STATES>(STATES, 0) = m_sqrtQ;
156
157 // Compute prior state covariance
158 // equation (5)
159 m_S = S_bar.householderQr().matrixQR().template block<STATES, STATES>(0, 0).template triangularView<Eigen::Upper>();

194 template <int ROWS>
195 void correct(
196 const EVec<ROWS> &y, const InputVector &u, const EMat<ROWS, STATES> &C, const EMat<ROWS, INPUTS> &D,
197 const EMat<ROWS, ROWS> &sqrtR
198 ) {
199 // R is sqrt(measurement noise)
200
201 // Compute measurement innovation
202 // equation (8)
203 EVec<ROWS> z = y - (C * m_xhat + D * u);
204
205 // Form temporary matrix to compute innovation covariance
206 EMat<STATES + ROWS, ROWS> Sy_bar;
207 Sy_bar.template block<STATES, ROWS>(0, 0) = (m_S * C.transpose());
208 Sy_bar.template block<ROWS, ROWS>(STATES, 0) = sqrtR;
209
210 // Compute innovation covariance
211 // equation (9)
212 EMat<ROWS, ROWS> Sy =
213 Sy_bar.householderQr().matrixQR().template block<ROWS, ROWS>(0, 0).template triangularView<Eigen::Upper>();
214
215 // Compute Kalman gain
216 // equation (10)
217 EMat<STATES, ROWS> K =
218 (Sy.template triangularView<Eigen::Upper>().solve(
219 (Sy.transpose().template triangularView<Eigen::Lower>().solve(H)) * (m_S.transpose() * m_S)
220 )).transpose();
221
222 // Compute posterior state
223 // equation (13)
224 m_xhat = m_xhat + K * z;
225
226 // Form temporary matrix to compute posterior state covariance
227 EMat<STATES + ROWS, STATES> S_bar;
228 S_bar.template block<STATES, STATES>(0, 0) = (m_S * (EMat<STATES, STATES>::Identity() - (K * H)).transpose());
229 S_bar.template block<ROWS, STATES>(STATES, 0) = (sqrtR * K.transpose());
230
231 // Compute posterior state covariance
232 // equation (14)
233 m_S = S_bar.householderQr().matrixQR().template block<STATES, STATES>(0, 0).template triangularView<Eigen::Upper>();
234 }
