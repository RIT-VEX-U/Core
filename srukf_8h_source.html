228 // equation (17)

231 // Project each sigma point forward in time according to the
232 // dynamics f(x, u)
233 //
234 // sigmas = 𝒳ₖ₋₁

331 template <int ROWS>
332 void correct(
333 const InputVector &u, const EVec<ROWS> &y,
334 const std::function<EVec<ROWS>(const StateVector &, const InputVector &)> &h,
335 const EVec<ROWS> measurement_stddevs,
336 const std::function<EVec<ROWS>(const EMat<ROWS, NUM_SIGMAS> &, const EVec<NUM_SIGMAS> &)> &mean_func_Y,
337 const std::function<EVec<ROWS>(const EVec<ROWS> &, const EVec<ROWS> &)> &residual_func_Y,
338 const std::function<StateVector(const StateVector &, const StateVector &)> &residual_func_X,
339 const std::function<StateVector(const StateVector &, const StateVector &)> &add_func_X
340 ) {
341
342 EMat<ROWS, ROWS> sqrt_R = measurement_stddevs.asDiagonal();
343
344 // Generate new sigma points from the prior mean and covariance
345 // and transform them into measurement space using h(x, u)
346 //
347 // sigmas = 𝒳
348 // sigmasH = 𝒴
349 //
350 // This differs from equation (22) which uses
351 // the prior sigma points, regenerating them allows
352 // multiple measurement updates per time update
353 EMat<ROWS, NUM_SIGMAS> sigmas_H;
354 EMat<STATES, NUM_SIGMAS> sigmas = m_pts.square_root_sigma_points(m_xhat, m_S);
355 for (int i = 0; i < m_pts.num_sigmas(); ++i) {
356 sigmas_H.template block<ROWS, 1>(0, i) = h(sigmas.template block<STATES, 1>(0, i), u);
357 }
358
359 // Pass the predicted measurement sigmas through the Unscented Transform
360 // to compute the mean predicted measurement and square-root innovation
361 // covariance.
362 //
363 // equations (23) (24) and (25)
364 auto [yhat, Sy] = square_root_ut<ROWS, STATES, NUM_SIGMAS>(
365 sigmas_H, m_pts.Wm(), m_pts.Wc(), mean_func_Y, residual_func_Y, sqrt_R.template triangularView<Eigen::Lower>()
366 );
367
368 // Compute cross covariance of the predicted state and measurement sigma
369 // points given as:
370 //
371 // L+1
372 // P_{xy} = Σ Wᵢ⁽ᶜ⁾[𝒳ᵢ - x̂][𝒴ᵢ - ŷ⁻]ᵀ
373 // i=0
374 //
375 // equation (26)
376 EMat<STATES, ROWS> Pxy;
377 Pxy.setZero();
378 for (int i = 0; i < m_pts.num_sigmas(); ++i) {
379 Pxy += m_pts.Wc(i) * (residual_func_X(m_sigmas_F.template block<STATES, 1>(0, i), m_xhat)) *
380 (residual_func_Y(sigmas_H.template block<ROWS, 1>(0, i), yhat)).transpose();
381 }
382
383 // Compute the Kalman gain. We use Eigen's forward and backward substitution
384 // to solve. The equation in the paper uses MATLAB's / operator, but Eigen's
385 // solvers act like the \ operator, so we need to rearrange the equation to
386 // use those.
387 //
388 // K = (P_{xy} / S_{y}ᵀ) / S_{y}
389 // K = (S_{y} \ P_{xy})ᵀ / S_{y}
390 // K = (S_{y}ᵀ \ (S_{y} \ P_{xy}ᵀ))ᵀ
391 //
392 // equation (27)
393 EMat<STATES, ROWS> K = (Sy.transpose().template triangularView<Eigen::Upper>().solve(
394 Sy.template triangularView<Eigen::Lower>().solve(Pxy.transpose())
395 ))
396 .transpose();
397
398 // Compute the posterior state mean
399 //
400 // x̂ = x̂⁻ + K(y − ŷ⁻)
401 //
402 // second part of equation (27)
403 m_xhat = add_func_X(m_xhat, K * residual_func_Y(y, yhat));
404
405 // Compute the intermediate matrix U for downdating
406 // the square-root covariance
407 //
408 // equation (28)
409 EMat<STATES, ROWS> U = K * Sy;
410
411 // Downdate the posterior square-root state covariance
412 //
413 // equation (29)
414 for (int i = 0; i < ROWS; i++) {
415 Eigen::internal::llt_inplace<double, Eigen::Lower>::rankUpdate(m_S, U.template block<STATES, 1>(0, i), -1);
416 }
417 }
418
419 private:
420 std::function<StateVector(const StateVector &, const InputVector &)> m_f;
421 std::function<OutputVector(const StateVector &, const InputVector &)> m_h;
422 std::function<StateVector(const EMat<STATES, NUM_SIGMAS> &, const EVec<NUM_SIGMAS> &)> m_mean_func_X;
423 std::function<OutputVector(const EMat<OUTPUTS, NUM_SIGMAS> &, const EVec<NUM_SIGMAS> &)> m_mean_func_Y;
424 std::function<StateVector(const StateVector &, const StateVector &)> m_residual_func_X;
425 std::function<OutputVector(const OutputVector &, const OutputVector &)> m_residual_func_Y;
426 std::function<StateVector(const StateVector &, const StateVector &)> m_add_func_X;
427 StateVector m_xhat;
428 StateMatrix m_S;
429 StateMatrix m_sqrt_Q;
430 EVec<OUTPUTS> m_measurement_stddevs;
431 EMat<STATES, NUM_SIGMAS> m_sigmas_F;
432
433 ScaledSphericalSimplexSigmaPoints<STATES> m_pts;
434
435 WithInputIntegrator m_integrator;

457template <int COV_DIM, int STATES, int NUM_SIGMAS>
458std::tuple<EVec<COV_DIM>, EMat<COV_DIM, COV_DIM>> square_root_ut(
459 const EMat<COV_DIM, NUM_SIGMAS> &sigmas, const EVec<NUM_SIGMAS> &Wm, const EVec<NUM_SIGMAS> &Wc,
460 const std::function<EVec<COV_DIM>(const EMat<COV_DIM, NUM_SIGMAS> &, const EVec<NUM_SIGMAS> &)> &mean_func,
461 const std::function<EVec<COV_DIM>(const EVec<COV_DIM> &, const EVec<COV_DIM> &)> &residual_func,
462 const EMat<COV_DIM, COV_DIM> &sqrt_R
463) {
464 // New mean is usually just the sum of the sigmas * weights:
465 //
466 // L+1
467 // x̂ = Σ Wᵢ⁽ᵐ⁾𝒳ᵢ
468 // i=0
469 //
470 // equations (19) and (23) in the paper show this,
471 // but we allow a custom function, usually for angle wrapping
472 EVec<COV_DIM> x = mean_func(sigmas, Wm);
473
474 // Form an intermediate matrix S⁻ as:
475 //
476 // [√{W₁⁽ᶜ⁾}(𝒳_{1:L+1} - x̂) √{Rᵛ}]
477 //
478 // the part of equations (20) and (24) within the "qr{}"
479 EMat<COV_DIM, NUM_SIGMAS - 1 + COV_DIM> S_bar;
480 for (int i = 0; i < NUM_SIGMAS - 1; i++) {
481 S_bar.template block<COV_DIM, 1>(0, i) =
482 std::sqrt(Wc[1]) * residual_func(sigmas.template block<COV_DIM, 1>(0, i + 1), x);
483 }
484 S_bar.template block<COV_DIM, COV_DIM>(0, NUM_SIGMAS - 1) = sqrt_R;
485
486 // Compute the square-root covariance of the sigma points.
487 //
488 // We transpose S⁻ first because we formed it by horizontally
489 // concatenating each part; it should be vertical so we can take
490 // the QR decomposition as defined in the "QR Decomposition" passage
491 // of section 3. "EFFICIENT SQUARE-ROOT IMPLEMENTATION"
492 //
493 // The resulting matrix R is the square-root covariance S, but it
494 // is upper triangular, so we need to transpose it.
495 //
496 // equations (20) and (24)
497 EMat<COV_DIM, COV_DIM> S = S_bar.transpose()
498 .householderQr()
499 .matrixQR()
500 .template block<COV_DIM, COV_DIM>(0, 0)
501 .template triangularView<Eigen::Upper>()
502 .transpose();
503
504 // Update or downdate the square-root covariance with (𝒳₀-x̂)
505 // depending on whether its weight (W₀⁽ᶜ⁾) is positive or negative.
506 //
507 // equations (21) and (25)
508 Eigen::internal::llt_inplace<double, Eigen::Lower>::rankUpdate(
509 S, residual_func(sigmas.template block<COV_DIM, 1>(0, 0), x), Wc[0]
510 );
511
512 return std::make_tuple(x, S);
513}

533template <int STATES> class ScaledSphericalSimplexSigmaPoints {
534 public:
535 static constexpr int NUM_SIGMAS = STATES + 2;
536
545 ScaledSphericalSimplexSigmaPoints(double alpha = 0.001, double beta = 2) { compute_weights(alpha, beta); }
546
550 int num_sigmas() { return NUM_SIGMAS; }
551
562 EMat<STATES, NUM_SIGMAS> square_root_sigma_points(const EVec<STATES> &x, const EMat<STATES, STATES> &S) {
563
564 EMat<STATES, NUM_SIGMAS> C = EMat<STATES, NUM_SIGMAS>::Zero();
565
566 for (int row = 0; row < STATES; row++) {
567 C.row(row).segment(1, row + 1).setConstant(-q(row) / (row + 1));
568 }
569 C.diagonal(2) = q;
570
571 EMat<STATES, NUM_SIGMAS> sigmas = S * C;
572 sigmas.colwise() += x;
573
574 return sigmas;
575 }
576
580 const EVec<NUM_SIGMAS> &Wm() const { return m_Wm; }
581
585 const EVec<NUM_SIGMAS> &Wc() const { return m_Wc; }
586
592 double Wm(int i) const { return m_Wm(i); }
593
599 double Wc(int i) const { return m_Wc(i); }
600
601 private:
602 EVec<NUM_SIGMAS> m_Wm;
603 EVec<NUM_SIGMAS> m_Wc;
604 EVec<STATES> q;
605
614 void compute_weights(double alpha, double beta) {
615 double c = 1 / (alpha * alpha * (STATES + 1));
616 m_Wm = EVec<NUM_SIGMAS>::Constant(c);
617 m_Wc = EVec<NUM_SIGMAS>::Constant(c);
618
619 m_Wm(0) = 1 - (1 / (alpha * alpha));
620 m_Wc(0) = 1 - (1 / (alpha * alpha)) + (1 - alpha * alpha + beta);
621
622 EVec<STATES> t;
623 for (int i = 0; i < STATES; i++) {
624 t(i) = i + 1;
625 }
626
627 q = alpha * ((t * (STATES + 1)).cwiseQuotient(t + EVec<STATES>::Ones())).cwiseSqrt();
628 }
