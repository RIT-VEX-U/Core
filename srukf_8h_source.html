337 template <int ROWS>
338 void correct(
339 const InputVector &u, const EVec<ROWS> &y,
340 const std::function<EVec<ROWS>(const StateVector &, const InputVector &)> &h,
341 const EVec<ROWS> measurement_stddevs,
342 const std::function<EVec<ROWS>(const EMat<ROWS, NUM_SIGMAS> &, const EVec<NUM_SIGMAS> &)> &mean_func_Y,
343 const std::function<EVec<ROWS>(const EVec<ROWS> &, const EVec<ROWS> &)> &residual_func_Y,
344 const std::function<StateVector(const StateVector &, const StateVector &)> &residual_func_X,
345 const std::function<StateVector(const StateVector &, const StateVector &)> &add_func_X
346 ) {
347
348 EMat<ROWS, ROWS> sqrt_R = measurement_stddevs.asDiagonal();
349
350 // Generate new sigma points from the prior mean and covariance
351 // and transform them into measurement space using h(x, u)
352 //
353 // sigmas = 𝒳
354 // sigmasH = 𝒴
355 //
356 // This differs from equation (22) which uses
357 // the prior sigma points, regenerating them allows
358 // multiple measurement updates per time update
359 EMat<ROWS, NUM_SIGMAS> sigmas_H;
360 EMat<STATES, NUM_SIGMAS> sigmas = m_pts.square_root_sigma_points(m_xhat, m_S);
361 for (int i = 0; i < NUM_SIGMAS; ++i) {
362 sigmas_H.template block<ROWS, 1>(0, i) =
363 h(sigmas.template block<STATES, 1>(0, i), u);
364 }
365
366 // Pass the predicted measurement sigmas through the Unscented Transform
367 // to compute the mean predicted measurement and square-root innovation
368 // covariance.
369 //
370 // equations (23) (24) and (25)
371 auto [yhat, Sy] = square_root_ut<ROWS, STATES, NUM_SIGMAS>(
372 sigmas_H, m_pts.Wm(), m_pts.Wc(), mean_func_Y, residual_func_Y,
373 sqrt_R.template triangularView<Eigen::Lower>());
374
375 // Compute cross covariance of the predicted state and measurement sigma
376 // points given as:
377 //
378 // L+1
379 // P_{xy} = Σ Wᵢ⁽ᶜ⁾[𝒳ᵢ - x̂][𝒴ᵢ - ŷ⁻]ᵀ
380 // i=0
381 //
382 // equation (26)
383 EMat<STATES, ROWS> Pxy;
384 Pxy.setZero();
385 for (int i = 0; i < NUM_SIGMAS; ++i) {
386 Pxy += m_pts.Wc(i) *
387 (residual_func_X(m_sigmas_F.template block<STATES, 1>(0, i),
388 m_xhat)) *
389 (residual_func_Y(sigmas_H.template block<ROWS, 1>(0, i), yhat))
390 .transpose();
391 }
392
393 // Compute the Kalman gain. We use Eigen's forward and backward substitution
394 // to solve. The equation in the paper uses MATLAB's / operator, but Eigen's
395 // solvers act like the \ operator, so we need to rearrange the equation to
396 // use those.
397 //
398 // K = (P_{xy} / S_{y}ᵀ) / S_{y}
399 // K = (S_{y} \ P_{xy})ᵀ / S_{y}
400 // K = (S_{y}ᵀ \ (S_{y} \ P_{xy}ᵀ))ᵀ
401 //
402 // equation (27)
403 EMat<STATES, ROWS> K = (Sy.transpose().template triangularView<Eigen::Upper>().solve(
404 Sy.template triangularView<Eigen::Lower>().solve(Pxy.transpose())
405 ))
406 .transpose();
407
408 // Compute the posterior state mean
409 //
410 // x̂ = x̂⁻ + K(y − ŷ⁻)
411 //
412 // second part of equation (27)
413 EVec<STATES> xhat_dot = K * residual_func_Y(y, yhat);
414 EVec<STATES> xhat = add_func_X(m_xhat, xhat_dot);
415 EMat<STATES, STATES> S = m_S;
416
417 // RECALIBRATE
418
419 // Add the change of xhat to each of the sigma points in 𝒳.
420 for (int i = 0; i < NUM_SIGMAS; i++) {
421 sigmas.template block<STATES, 1>(0, i) += (xhat_dot);
422 }
423
424 // Pass those sigma points through the measurement function to transform
425 // them into measurement space.
426 for (int i = 0; i < NUM_SIGMAS; ++i) {
427 sigmas_H.template block<ROWS, 1>(0, i) = h(sigmas.template block<STATES, 1>(0, i), u);
428 }
429
430 // Perform a second unscented transform, this time on the recalibrated
431 // measurement sigma points.
432 auto [yhat_k, Sy_k] = square_root_ut<ROWS, STATES, NUM_SIGMAS>(
433 sigmas_H, m_pts.Wm(), m_pts.Wc(), mean_func_Y, residual_func_Y,
434 sqrt_R.template triangularView<Eigen::Lower>());
435
436 // Compute the cross covariance of the recalibrated sigma points.
437 Pxy.setZero();
438 for (int i = 0; i < NUM_SIGMAS; ++i) {
439 Pxy += m_pts.Wc(i) *
440 (residual_func_X(sigmas.template block<STATES, 1>(0, i),
441 xhat)) *
442 (residual_func_Y(sigmas_H.template block<ROWS, 1>(0, i), yhat_k))
443 .transpose();
444 }
445
446 // Compute the intermediate matrix U for downdating
447 // the square-root covariance
448 //
449 // equation (28)
450 const EMat<STATES, ROWS> U = K * Sy;
451
452 // Downdate the posterior square-root state covariance
453 //
454 // equation (29)
455 for (int i = 0; i < ROWS; i++) {
456 Eigen::internal::llt_inplace<double, Eigen::Lower>::rankUpdate(
457 S, U.template block<STATES, 1>(0, i), -1);
458 }
459
460 // BACK OUT
461
462 // We only use the posterior state and covariance if it is more certain
463 // than the prior.
464 if (m_S.trace() > S.trace()) {
465 m_xhat = xhat;
466 m_S = S;
467 }
468 }

470 private:
471 std::function<StateVector(const StateVector &, const InputVector &)> m_f;
472 std::function<OutputVector(const StateVector &, const InputVector &)> m_h;
473 std::function<StateVector(const EMat<STATES, NUM_SIGMAS> &, const EVec<NUM_SIGMAS> &)> m_mean_func_X;
474 std::function<OutputVector(const EMat<OUTPUTS, NUM_SIGMAS> &, const EVec<NUM_SIGMAS> &)> m_mean_func_Y;
475 std::function<StateVector(const StateVector &, const StateVector &)> m_residual_func_X;
476 std::function<OutputVector(const OutputVector &, const OutputVector &)> m_residual_func_Y;
477 std::function<StateVector(const StateVector &, const StateVector &)> m_add_func_X;
478 StateVector m_xhat;
479 StateMatrix m_S;
480 StateMatrix m_sqrt_Q;
481 EVec<OUTPUTS> m_measurement_stddevs;

508template <int COV_DIM, int STATES, int NUM_SIGMAS>
509std::tuple<EVec<COV_DIM>, EMat<COV_DIM, COV_DIM>> square_root_ut(
510 const EMat<COV_DIM, NUM_SIGMAS> &sigmas, const EVec<NUM_SIGMAS> &Wm, const EVec<NUM_SIGMAS> &Wc,
511 const std::function<EVec<COV_DIM>(const EMat<COV_DIM, NUM_SIGMAS> &, const EVec<NUM_SIGMAS> &)> &mean_func,
512 const std::function<EVec<COV_DIM>(const EVec<COV_DIM> &, const EVec<COV_DIM> &)> &residual_func,
513 const EMat<COV_DIM, COV_DIM> &sqrt_R
514) {
515 // New mean is usually just the sum of the sigmas * weights:
516 //
517 // L+1
518 // x̂ = Σ Wᵢ⁽ᵐ⁾𝒳ᵢ
519 // i=0
520 //
521 // equations (19) and (23) in the paper show this,
522 // but we allow a custom function, usually for angle wrapping
523 EVec<COV_DIM> x = mean_func(sigmas, Wm);
524
525 // Form an intermediate matrix S⁻ as:
526 //
527 // [√{W₁⁽ᶜ⁾}(𝒳_{1:L+1} - x̂) √{Rᵛ}]
528 //
529 // the part of equations (20) and (24) within the "qr{}"
530 EMat<COV_DIM, NUM_SIGMAS - 1 + COV_DIM> S_bar;
531 for (int i = 0; i < NUM_SIGMAS - 1; i++) {
532 S_bar.template block<COV_DIM, 1>(0, i) =
533 std::sqrt(Wc[1]) * residual_func(sigmas.template block<COV_DIM, 1>(0, i + 1), x);
534 }
535 S_bar.template block<COV_DIM, COV_DIM>(0, NUM_SIGMAS - 1) = sqrt_R;
536
537 // Compute the square-root covariance of the sigma points.
538 //
539 // We transpose S⁻ first because we formed it by horizontally
540 // concatenating each part; it should be vertical so we can take
541 // the QR decomposition as defined in the "QR Decomposition" passage
542 // of section 3. "EFFICIENT SQUARE-ROOT IMPLEMENTATION"
543 //
544 // The resulting matrix R is the square-root covariance S, but it
545 // is upper triangular, so we need to transpose it.
546 //
547 // equations (20) and (24)
548 EMat<COV_DIM, COV_DIM> S = S_bar.transpose()
549 .householderQr()
550 .matrixQR()
551 .template block<COV_DIM, COV_DIM>(0, 0)
552 .template triangularView<Eigen::Upper>()
553 .transpose();
554
555 // Update or downdate the square-root covariance with (𝒳₀-x̂)
556 // depending on whether its weight (W₀⁽ᶜ⁾) is positive or negative.
557 //
558 // equations (21) and (25)
559 Eigen::internal::llt_inplace<double, Eigen::Lower>::rankUpdate(
560 S, residual_func(sigmas.template block<COV_DIM, 1>(0, 0), x), Wc[0]
561 );
562
563 return std::make_tuple(x, S);
564}

585 public:
586 static constexpr int NUM_SIGMAS = STATES + 2;

596 ScaledSphericalSimplexSigmaPoints(double alpha = 0.001, double beta = 2) { compute_weights(alpha, beta); }

601 int num_sigmas() { return NUM_SIGMAS; }

613 EMat<STATES, NUM_SIGMAS> square_root_sigma_points(const EVec<STATES> &x, const EMat<STATES, STATES> &S) {
614
615 EMat<STATES, NUM_SIGMAS> C = EMat<STATES, NUM_SIGMAS>::Zero();
616
617 for (int row = 0; row < STATES; row++) {
618 C.row(row).segment(1, row + 1).setConstant(-q(row) / (row + 1));
619 }
620 C.diagonal(2) = q;
621
622 EMat<STATES, NUM_SIGMAS> sigmas = S * C;
623 sigmas.colwise() += x;
624
625 return sigmas;
626 }
627
631 const EVec<NUM_SIGMAS> &Wm() const { return m_Wm; }
632
636 const EVec<NUM_SIGMAS> &Wc() const { return m_Wc; }
637
643 double Wm(int i) const { return m_Wm(i); }
644
650 double Wc(int i) const { return m_Wc(i); }
651
652 private:
653 EVec<NUM_SIGMAS> m_Wm;
654 EVec<NUM_SIGMAS> m_Wc;
655 EVec<STATES> q;
656
665 void compute_weights(double alpha, double beta) {
666 double c = 1 / (alpha * alpha * (STATES + 1));
667 m_Wm = EVec<NUM_SIGMAS>::Constant(c);
668 m_Wc = EVec<NUM_SIGMAS>::Constant(c);
669
670 m_Wm(0) = 1 - (1 / (alpha * alpha));
671 m_Wc(0) = 1 - (1 / (alpha * alpha)) + (1 - alpha * alpha + beta);
672
673 EVec<STATES> t;
674 for (int i = 0; i < STATES; i++) {
675 t(i) = i + 1;
676 }
677
678 q = alpha * ((t * (STATES + 1)).cwiseQuotient(t + EVec<STATES>::Ones())).cwiseSqrt();
679 }
